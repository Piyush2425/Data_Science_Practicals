# IQR method
Q1 = df['reading_score'].quantile(0.25)
Q3 = df['reading_score'].quantile(0.75)
IQR = Q3 - Q1

lower = Q1 - 1.5 * IQR
upper = Q3 + 1.5 * IQR

outliers = df[(df['reading_score'] < lower) | (df['reading_score'] > upper)]
